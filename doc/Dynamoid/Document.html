def ==(other)
  return false if other.nil?
  other.respond_to?(:id) && other.id == self.id
end

def initialize(attrs = {})
  @new_record = true
  @attributes ||= {}
  attrs = self.class.undump(attrs)
  self.class.attributes.keys.each {|att| write_attribute(att, attrs[att])}
end
